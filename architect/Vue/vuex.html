# Vuex

Vuex是什么？ 简单来说：对Vue应用中多个组件的共享状态进行集中式的管理

安装

    npm install vuex --save

vuex是vue中的一个组件 在使用时，需要导入，在main.js中显示的声明使用该插件 Vue.use(Vuex);

代码实例

//使用Vuex做一个例子
    
import Vue from 'vue'
import Vuex from 'vuex'


Vue.use(Vuex);


//状态对象
const state = {
    count:0        //初始的状态值
};
//包含多个更新state函数的对象
const mutations = {    //commit之后具体修改state的方法
    ExampleMEthod1(state){
        //具体的修改state值
    },
    ExampleMethod2(state){
        //
    }
};
//包含多个对应事件回调函数的对象
const actions = {
    example({commit,state},data){    //由前台dispatch分发action到此执行该action  this.$store.dispatch('eample');
        commit('ExampleMethod1'); //命名提交
    },
    exampleAsync({commit}){    //在action还可以直接执行异步的代码
        setTimeout({()=>{
            commit('ExampleMethod2')
            }
        },1000);
    }
};
//包含多个getter计算属性函数的对象
const getters = {
    读取属性值
    exampleGetter(state){
        return state.count    //这里可以初次判断回显到页面 this.$store.getters.exampleGetter; 不用加括号，加括号表示调用方法，我们只是为了读取属性值
    }
};


export default new Vuex.Store({
    state, //状态对象
    mutations, //包含多个更新state函数的对象
    actions, //包含多个对应事件回调函数的对象
    getters //包含多个getter计算属性函数的对象
});

//实际的操作中，可能各个vuex的state、mutations、actions、getters会分开写


//回显数据页面可以使用vuex自带的API
import {mapActions,mapGetters,mapState} form 'vuex';

//render配置：
//在main.js中还可以这样加入APP组件
new Vue({
    el:'#app',
    render:h => h(App)  // => 表示执行其后面的函数并且带上一个返回值
});
